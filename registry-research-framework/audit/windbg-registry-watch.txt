$$ RegProbe WinDbg boot registry watch script
$$ This script logs CmQueryValueKey value-name probes and relies on post-filtering for the target names below.
.symfix
.reload /f
.logopen /t "C:\\RegProbe-Diag\\windbg-registry-trace.log"
.echo REGPROBE_WATCH_KEYS_BEGIN
.echo AllowAudioToEnableExecutionRequiredPowerRequests
.echo AllowSystemRequiredPowerRequests
.echo AlwaysComputeQosHints
.echo CoalescingFlushInterval
.echo IdleProcessorsRequireQosManagement
.echo REGPROBE_WATCH_KEYS_END
bc *
bu nt!CmQueryValueKey
bs 0 ".echo REGPROBE_VALUE_BEGIN; du poi(@rdx+8); .echo REGPROBE_VALUE_END; gc"
g
